Re-size 動画を保存
動画をリサイズします(FHD) ※ファイルを選択して画像が表示されてから”Re-size"を一回押してください、ボタンの色が元に戻ると"動画を保存”ボタンが表示されます、ボタンを押すと保存されます。元の動画には影響ありません。
何かエラーが出たらブラウザを更新してください。
※PCは"応答がありません”とメッセージが表示されますが、放置してかまいません、終了したらメッセージは消えます
※処理後の動画ファイルに音声はありません！！(そのうち対応するかも)
{ "packages" : ["opencv-python","numpy"] } import cv2 as cv import numpy as np import base64 from io import BytesIO import sys import tempfile from pyscript import when, display from pyscript.web import img, page, video, a b64_header='data:video/mp4;base64,' @when("click", "#button") def click_handler(event): my_input=page["video"] img_binary = base64.b64decode(my_input[0].src[len(b64_header):]) img_b64 = mresize(img_binary) dl = page['a'] dl[1].href = b64_header + img_b64 dl[1].style = "display:inline" def mresize(bdata): with tempfile.NamedTemporaryFile(mode='wb') as tmpf: tmpf.write(bdata) cap = cv.VideoCapture(tmpf.name) width = int(cap.get(cv.CAP_PROP_FRAME_WIDTH)) height = int(cap.get(cv.CAP_PROP_FRAME_HEIGHT)) fps = cap.get(cv.CAP_PROP_FPS) ratio = 1080/height fourcc = cv.VideoWriter_fourcc('m','p','4','v') with tempfile.TemporaryDirectory() as tmpd: writer = cv.VideoWriter(tmpd + '/tmp.mp4', fourcc, fps, (int(width * ratio), int(height * ratio))) while True: ret, frame = cap.read() if not ret: break frame = cv.resize(frame, (int(width * ratio), int(height * ratio))) writer.write(frame) writer.release() cap.release() with open(tmpd + '/tmp.mp4', mode='rb') as fh: buf = fh.read() v_out = base64.b64encode(buf).decode('ascii') return v_out
